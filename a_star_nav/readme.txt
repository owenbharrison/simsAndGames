navigation
mouse middle - zoom
mouse left - pan
home - zoom to fit
shift - snap mouse to grid

render modes
g - toggle grid
b - toggle background
i - toggle ids

editing
mouse left - drag
mouse right - select
a - add node
c - connect nodes
x - delete
ctrl c - copy
ctrl v - paste
ctrl a - select all

routing
f - set from waypoint
t - set to waypoint
tab - swap waypoints
enter - route

misc
esc - console

commands:
import filename
export filename
reset
clear
